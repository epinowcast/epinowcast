28vector diagSPD_EQ(real alpha, real rho, real L, int M) {
29 vector[M] indices = linspaced_vector(M, 1, M);
30 real factor = alpha * sqrt(sqrt(2 * pi()) * rho);
31 real exponent = -0.25 * (rho * pi() / 2 / L)^2;

54vector diagSPD_Matern32(real alpha, real rho, real L, int M) {
55 real factor = 2 * alpha * (sqrt(3) / rho)^1.5;
56 vector[M] denom = 3 / square(rho) + matern_indices(M, L);

63vector diagSPD_Matern52(real alpha, real rho, real L, int M) {
64 real factor = 16 * pow(sqrt(5) / rho, 5);
65 vector[M] denom = 3 * pow(5 / square(rho) + matern_indices(M, L), 3);

72vector diagSPD_Periodic(real alpha, real rho, int M) {
73 real a = inv_square(rho);
74 vector[M] indices = linspaced_vector(M, 1, M);
75 vector[M] q = exp(
76 log(alpha) + 0.5 *
77 (log(2) - a + to_vector(log_modified_bessel_first_kind(indices, a)))
78 );
79 return append_row(q, q);

95vector update_gp(matrix PHI, int M, real L, real alpha,
96 real rho, vector eta, int type, real nu) {
97 vector[type == 1 ? 2 * M : M] diagSPD;
98 if (type == 0) {
99 diagSPD = diagSPD_EQ(alpha, rho, L, M);
100 } else if (type == 1) {
101 diagSPD = diagSPD_Periodic(alpha, rho, M);
102 } else if (type == 2) {
103 if (nu == 0.5) {
104 diagSPD = diagSPD_Matern12(alpha, rho, L, M);
105 } else if (nu == 1.5) {
106 diagSPD = diagSPD_Matern32(alpha, rho, L, M);
107 } else if (nu == 2.5) {
108 diagSPD = diagSPD_Matern52(alpha, rho, L, M);
109 } else {
110 reject("nu must be one of 0.5, 1.5, or 2.5; found nu=", nu);
111 }
112 }
113 return PHI * (diagSPD .* eta);
114}

169matrix gp_latent_matrix(int T, int G, int M, real L, int type, real nu,
170 int d, matrix PHI, matrix eta,
171 array[] real rho, array[] real alpha) {
172 matrix[T, G] gp_eps;
173 for (g in 1:G) {
174 vector[rows(PHI)] f = update_gp(
175 PHI, M, L, alpha[1], rho[1], eta[, g], type, nu
176 );
177 if (d == 0) {
178 gp_eps[, g] = f;
179 } else {
180 vector[T] col = rep_vector(0.0, T);
181 col[(d + 1):T] = f;
182 for (i in 1:d) {
183 col = cumulative_sum(col);
184 }
185 gp_eps[, g] = col;
186 }
187 }
188 return gp_eps;
189}

213vector apply_gp_term(vector base, int present, int centre, int T, int G,
214 int M, real L, int type, real nu, int d,
215 matrix PHI, matrix eta,
216 array[] real rho, array[] real alpha,
217 array[] int flat_idx) {
218 if (!present) return base;
219 matrix[T, G] gp_eps = gp_latent_matrix(
220 T, G, M, L, type, nu, d, PHI, eta, rho, alpha
221 );
222 // Centre the integrated GP (d >= 1) on its grand mean so the intercept
223 // owns the level (mirrors the ARIMA residual). Only the shared grand
224 // mean is removed, so a grouped GP keeps each group's own level.
225 if (centre && d >= 1) {
226 gp_eps = gp_eps - mean(to_vector(gp_eps));
227 }
228 return base + to_vector(gp_eps)[flat_idx];
229}
